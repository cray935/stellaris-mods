
######################## Create Species ##########################################
#
# Some effects only work in planet scope. 
# The current planet will be the homeworld of the species created.
#
##################################################################################

len_create_species = {
	log = "start len_create_species"
	if = {
		limit = { has_planet_flag = len_species_random }
		len_create_random = yes
	}
	else_if = {
		limit = { has_planet_flag = len_species_mam }
		len_create_mam = yes
	}
	else_if = {
		limit = { has_planet_flag = len_species_rep }
		len_create_rep = yes
	}
	else_if = {
		limit = { has_planet_flag = len_species_avi }
		len_create_avi = yes
	}
	else_if = {
		limit = { has_planet_flag = len_species_art }
		len_create_art = yes
	}
	else_if = {
		limit = { has_planet_flag = len_species_mol }
		len_create_mol = yes
	}
	else_if = {
		limit = { has_planet_flag = len_species_fun }
		len_create_fun = yes
	}
	else_if = {
		limit = { has_planet_flag = len_species_plant }
		len_create_plant = yes
	}
	else_if = {
		limit = { has_planet_flag = len_species_hum }
		len_create_hum = yes
	}
	
	len_cleanup_species_flags = yes
	
	last_created_species = { 
		log = "last_created_species"
		print_scope_effect = yes
		save_global_event_target_as = len_species
	}
	event_target:len_species = {
		log = "event_target:len_species"
		print_scope_effect = yes
	}
	save_global_event_target_as = len_species_planet
	log = "stop len_create_species"
}

len_create_random = {
	if = {
		limit = { has_planet_flag = len_species_sentient }
		
		if = {
			limit = { has_planet_flag = len_species_random_traits }
			
			create_species = {
				name = random
				class = random
				portrait = random
				homeworld = THIS
				traits = {
					trait = len_trait_engineered
					trait = random_traits
				}
			}
		}
		else = {
			create_species = {
				name = random
				class = random
				portrait = random
				homeworld = THIS
				traits = {
					trait = len_trait_engineered
				}
			}
		}
	}
	else = {
		create_species = {
			name = random
			class = random
			portrait = random
			homeworld = THIS
			traits = {
				trait = len_trait_engineered
				trait = random_presapient_trait
			}
			sapient = no
		}
	}
}
