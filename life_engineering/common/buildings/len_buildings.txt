# Promethean Project
len_building_lef_project_promethean = {

	base_buildtime = 500
	planet_unique = yes
	is_listed = no
	
	cost = {
		minerals = 1000
		energy = 550
	}
	
	produced_resources = {
		society_research = 4
	}
	
	required_resources = {
		energy = 2
	}
	
	potential = {
		always = yes
	}
	
	allow = {
		custom_tooltip = {
			text = "len_promethian_project_requirement_planet"
			planet = {
				count_pops = {
					count > 9
					limit = {
						is_colony_pop = yes
						is_robot_pop = no
					}
				}
			}
		}
		
		custom_tooltip = {
			text = "len_promethian_project_requirement_tile"
			tile = {
				has_grown_pop = yes
				pop = { is_robot_pop = no }
			}
		}
	}
	
	prerequisites = {
		"len_tech_promethean_project"
	}
	
	ai_replace = no
}
